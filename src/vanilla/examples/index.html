Alert (JavaScript)

// Info Alert
Bigtablet.Alert({
  title: '알림',
  message: '작업이 완료되었습니다.',
  variant: 'info'
});

// Confirm Alert
Bigtablet.Alert({
  title: '확인',
  message: '정말 삭제하시겠습니까?',
  variant: 'warning',
  showCancel: true,
  onConfirm: () => console.log('Confirmed!'),
  onCancel: () => console.log('Cancelled')
});
